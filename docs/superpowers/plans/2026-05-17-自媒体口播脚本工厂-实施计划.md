```python
"""
Agent 之间传递的 Pydantic 数据模型。
所有 Agent 的输入输出都通过这里定义的结构体传递，
确保类型安全和格式一致。

from pydantic import BaseModel, Field


class Section(BaseModel):
    """大纲中的一个段落。"""
    heading: str = Field(..., description="段落标题")
    talking_points: list[str] = Field(default_factory=list, description="要点列表")


class Outline(BaseModel):
    """策划 Agent 产出的大纲。"""
    title: str = Field(..., description="脚本标题")
    hook: str = Field(..., description="开头钩子，用于吸引观众注意")
    sections: list[Section] = Field(..., description="大纲段落列表")
    key_phrases: list[str] = Field(default_factory=list, description="金句列表，不少于3条")
    estimated_duration: int = Field(..., description="预估总时长(秒)")


class Script(BaseModel):
    """写作 Agent 产出的口播脚本。"""
    content: str = Field(..., description="口播稿全文")
    duration_estimate: int = Field(..., description="预估时长(秒)")
    tone_markers: list[str] = Field(default_factory=list, description="节奏标记: 快/慢/重音等")


class Issue(BaseModel):
    """审核 Agent 发现的问题。"""
    severity: str = Field(..., description="严重级别: P0=合规(必改) P1=事实(必改) P2=风格(仅记录)")

    location: str = Field(..., description="出问题的原文片段")

class ReviewResult(BaseModel):

    score: float = Field(..., description="综合评分 0-100")
    dimension_scores: dict = Field(

            "information": 0,
            "oral": 0,
            "compliance": 0,
            "usability": 0,
```

```python
"""
Agent 基类。
所有 Agent 继承此类，遵守统一接口约定：
输入 state dict → 调 LLM → 返回更新后的 state dict。
"""

from abc import ABC, abstractmethod
from langchain_openai import ChatOpenAI
from backend.app.core.config import get_settings


class BaseAgent(ABC):
    """Agent 基类。

    每个 Agent 只做三件事：
    1. 从 state 读自己需要的数据
    2. 构造 prompt → 调 LLM → 解析结构化输出
    3. 把结果写回 state 并返回

    Agent 不知道前一个是谁，也不知道下一步去哪。
    编排器(pipeline)负责 step 流转。
    """

    def __init__(self, model_name: str | None = None):
        """
        初始化 Agent。

        Args:
            model_name: DeepSeek 模型名。不传则用 deepseek-chat。
        """
        settings = get_settings()
        self.model_name = model_name or settings.deepseek_chat_model
        self.llm = ChatOpenAI(
            model=self.model_name,
            openai_api_key=settings.deepseek_api_key,
            openai_api_base=settings.deepseek_base_url,
            temperature=0.7,
            max_tokens=4096,
        )

    @abstractmethod
    def run(self, state: dict) -> dict:
        """
        执行 Agent 的核心逻辑。

        Args:
            state: 当前流水线状态字典

        Returns:
            更新后的状态字典（在原 state 上修改并返回）
        """
        pass
```

```python
"""
策划 Agent。
职责：分析选题 → 搜索热点 → 输出大纲 + 金句。
使用 deepseek-reasoner（推理能力强），接入 Tavily 搜索。
"""

import json
from langchain_core.prompts import ChatPromptTemplate
from langchain_core.output_parsers import PydanticOutputParser
from backend.app.agents.base import BaseAgent
from backend.app.schemas.agent import Outline, Section
from backend.app.core.config import get_settings


class PlanningAgent(BaseAgent):
    """策划 Agent —— MCN 资深编导角色。"""

    def __init__(self):
        # 策划需要推理能力，用 deepseek-reasoner
        super().__init__(model_name=get_settings().deepseek_reasoner_model)
        self.parser = PydanticOutputParser(pydantic_object=Outline)

    def _search_hotspot(self, topic: str) -> str:
        """用 Tavily 搜索选题相关热点素材。

        Args:
            topic: 选题关键词

        Returns:
            搜索结果的文本摘要
        """
        settings = get_settings()
        if not settings.tavily_api_key:
            return ""  # 无 API key 时跳过搜索

        try:
            from tavily import TavilyClient
            client = TavilyClient(api_key=settings.tavily_api_key)
            result = client.search(
                query=f"{topic} 热门话题 短视频",
                search_depth="basic",
                max_results=3,
            )
            summaries = [r.get("content", "")[:300] for r in result.get("results", [])]
            return "\n".join(summaries) if summaries else ""
        except Exception:
            return ""  # 搜索失败不阻塞主流程

    def _build_prompt(self, topic: str, style: str, duration: int,
                      feedback: str = "", hotspot: str = ""):
        """构造策划 prompt。

        Args:
            topic: 选题主题
            style: 账号风格（知识/搞笑/情感）
            duration: 目标时长(秒)
            feedback: 用户的修改反馈（修改大纲时使用）
            hotspot: 搜索到的热点信息

        Returns:
            ChatPromptTemplate 对象
        """
        format_instructions = self.parser.get_format_instructions()
        feedback_text = f"\n修改要求：{feedback}" if feedback else ""
        hotspot_text = f"\n热点参考：\n{hotspot}" if hotspot else ""

        system_prompt = f"""你是一个MCN机构的资深编导，有8年短视频口播策划经验。
你的任务是为博主策划一个口播视频大纲。

账号风格：{style}
目标时长：{duration}秒（约{duration // 2}字口语文本）
{hotspot_text}
{feedback_text}

策划要求：
1. 开头钩子要在3秒内抓住观众注意力
2. 大纲结构清晰：引入→展开→高潮→收尾
3. 至少产出3条金句，可用于结尾引发转赞评
4. 每个段落的 talking_points 是具体要说的话，不是抽象描述
5. {style}风格的账号，用{style}风格的语气和节奏

{format_instructions}"""

        return ChatPromptTemplate.from_messages([
            ("system", system_prompt),
            ("human", "选题：{topic}\n请输出结构化的口播大纲。"),
        ])

    def run(self, state: dict) -> dict:
        """执行策划逻辑。

        Args:
            state: 当前流水线状态

        Returns:
            更新了 outline 字段后的 state
        """
        topic = state.get("topic", "")
        style = state.get("style", "知识")
        duration = state.get("duration", 120)
        feedback = state.get("feedback", "")

        # 搜热点
        hotspot = self._search_hotspot(topic)

        # 调 LLM
        prompt = self._build_prompt(topic, style, duration, feedback, hotspot)
        chain = prompt | self.llm
        response = chain.invoke({"topic": topic})

        # 解析结构化输出
        try:
            outline = self.parser.parse(response.content)
        except Exception:
            # 解析失败时用原始文本做一个简化版 Outline
            outline = Outline(
                title=topic,
                hook="",
                sections=[Section(heading="内容", talking_points=[response.content[:200]])],
                key_phrases=[],
                estimated_duration=duration,
            )

        # 写回 state
        state["outline"] = outline.model_dump()
        return state
```

```python
        super().__init__()  # 默认 deepseek-chat，快且便宜
        self.parser = PydanticOutputParser(pydantic_object=Script)

    def _build_prompt(self, outline: dict, style: str, duration: int):
        """构造写作 prompt。

        Args:
            outline: 策划产出的大纲字典
            style: 账号风格
            duration: 目标时长(秒)

        Returns:
            ChatPromptTemplate 对象
        """
        format_instructions = self.parser.get_format_instructions()
        style_guide = STYLE_GUIDES.get(style, STYLE_GUIDES["知识"])
        word_count = duration // 2  # 口语约每秒2字

        # 把大纲转成可读文本
        sections_text = ""
        for i, section in enumerate(outline.get("sections", []), 1):
            heading = section.get("heading", "")
            points = "；".join(section.get("talking_points", []))
            sections_text += f"\n第{i}段: {heading}\n要点: {points}"

        system_prompt = f"""你是一个短视频口播脚本写手，专为{style}类账号写稿。
口播脚本要求：
1. 纯口语化写作，每句话都能读出嘴——短句、自然停顿、不堆长定语
2. 按{style}风格指南写：{style_guide}
3. 总字数约{word_count}字，时长约{duration}秒
4. 节奏标记用方括号标注：[快]/[慢]/[重音]/[停顿1s]
5. 目标：观众听完有获得感，愿意转赞评

{format_instructions}"""

        return ChatPromptTemplate.from_messages([
            ("system", system_prompt),
            ("human", "大纲：\n标题：{title}\n钩子：{hook}\n{sections}\n金句：{phrases}\n\n请按以上大纲写完整口播脚本。"),
        ])

    def run(self, state: dict) -> dict:
        """执行写作逻辑。

        Args:
            state: 当前流水线状态

        Returns:
            更新了 script 字段后的 state
        """
        outline = state.get("outline", {})
        style = state.get("style", "知识")
        duration = state.get("duration", 120)

        prompt = self._build_prompt(outline, style, duration)
        chain = prompt | self.llm
        response = chain.invoke({
            "title": outline.get("title", ""),
            "hook": outline.get("hook", ""),
            "sections": "",  # sections already in system prompt
            "phrases": "；".join(outline.get("key_phrases", [])),
        })

        # 解析结构化输出
        try:
            script = self.parser.parse(response.content)
        except Exception:
            script = Script(
                content=response.content,
                duration_estimate=duration,
                tone_markers=[],
            )

        state["script"] = script.model_dump()
        return state
```

```python
"""
审核 Agent。
职责：读脚本 → 四维审核 → 输出 issues(P0/P1/P2) + 评分。
使用 deepseek-reasoner（需要细致分析）。

from langchain_core.prompts import ChatPromptTemplate
from langchain_core.output_parsers import PydanticOutputParser
from backend.app.agents.base import BaseAgent
from backend.app.schemas.agent import ReviewResult, Issue
from backend.app.core.config import get_settings


class ReviewAgent(BaseAgent):
    """审核 Agent —— 四维审核 + Issue 分级。"""

    def __init__(self):
        super().__init__(model_name=get_settings().deepseek_reasoner_model)
        self.parser = PydanticOutputParser(pydantic_object=ReviewResult)

    def _fact_check(self, content: str) -> str:
        """用 Tavily 做事实核查。

        Args:
            content: 脚本全文

        Returns:
            搜索结果或空字符串
        """
        settings = get_settings()
        if not settings.tavily_api_key:
            return ""

        try:
            from tavily import TavilyClient
            client = TavilyClient(api_key=settings.tavily_api_key)
            # 抽取脚本中的关键事实声明做核查
            result = client.search(
                query=f"事实核查: {content[:200]}",
                search_depth="basic",
                max_results=2,
            )
            summaries = [r.get("content", "")[:200] for r in result.get("results", [])]
            return "\n".join(summaries) if summaries else ""
        except Exception:
            return ""

    def _build_prompt(self, content: str, style: str, fact_check: str):
        """构造审核 prompt。

        Args:
            content: 脚本全文
            style: 账号风格
            fact_check: 事实核查结果

        Returns:
            ChatPromptTemplate 对象
        """
        format_instructions = self.parser.get_format_instructions()
        fact_text = f"\n事实核查参考：\n{fact_check}" if fact_check else ""

        system_prompt = f"""你是一个严格的短视频脚本审核编辑。审核4个维度：

1. 信息量(25分)：有没有实质性内容？是干货还是空洞口号？
2. 口语化(25分)：读出来是否自然？有没有书面语长句？
3. 合规性(25分)：有没有广告法极限词（"最好""第一""国家级""唯一""顶级"）？有没有医疗健康类敏感断言？
4. 可用率(25分)：编导拿来能不能直接拍？缺不缺口播提示？

账号风格：{style}
{fact_text}

Issue 分级：
- P0 = 合规/法律问题，必须修改。如极限词、医疗断言、政治敏感
- P1 = 事实/数据错误，必须修改。如引用的数据与事实核查结果不符
- P2 = 风格/表达建议，仅记录。如"建议这句口语化一点"

passed=true 的条件：没有 P0 和 P1 问题。

{format_instructions}"""

        return ChatPromptTemplate.from_messages([
            ("system", system_prompt),
            ("human", "请审核以下脚本：\n\n{content}"),
        ])

    def run(self, state: dict) -> dict:
        """执行审核逻辑。

        Args:
            state: 当前流水线状态

        Returns:
            更新了 review 字段后的 state
        """
        script = state.get("script", {})
        content = script.get("content", "")
        style = state.get("style", "知识")

        # 事实核查
        fact_check = self._fact_check(content)

        # 调 LLM 做四维审核
        prompt = self._build_prompt(content, style, fact_check)
        chain = prompt | self.llm
        response = chain.invoke({"content": content})

        # 解析结构化输出
        try:
            review = self.parser.parse(response.content)
        except Exception:
            review = ReviewResult(
                passed=False,
                issues=[Issue(
                    severity="P0",
                    category="system",
                    location="",
                    description="审核Agent解析失败，请人工审核",
                    suggestion="手动检查脚本内容",
                )],
                score=0,
            )

        state["review"] = review.model_dump()
        return state
```

```python
"""
修改 Agent。
职责：读脚本 + issues → 逐条修改 P0/P1 → 输出新脚本。
使用 deepseek-chat（快，执行型任务不需要深度推理）。

from langchain_core.prompts import ChatPromptTemplate
from langchain_core.output_parsers import PydanticOutputParser
from backend.app.agents.base import BaseAgent
from backend.app.schemas.agent import Script


class RevisionAgent(BaseAgent):
    """修改 Agent —— 只改 P0/P1，保留原文风格。"""

    def __init__(self):
        super().__init__()  # deepseek-chat
        self.parser = PydanticOutputParser(pydantic_object=Script)

    def _build_prompt(self, content: str, must_fix_issues: list[dict], style_note_issues: list[dict]):
        """构造修改 prompt。

        Args:
            content: 原始脚本全文
            must_fix_issues: P0/P1问题列表（必须改）
            style_note_issues: P2问题列表（仅参考，不必改）

        Returns:
            ChatPromptTemplate 对象
        """
        format_instructions = self.parser.get_format_instructions()
        original_duration = len(content) // 2

        # 构造必须修改的清单
        fix_list = ""
        for i, issue in enumerate(must_fix_issues, 1):
            fix_list += f"\n{i}. 【{issue.get('severity','')}】位置: {issue.get('location','')}\n"
            fix_list += f"   问题: {issue.get('description','')}\n"
            fix_list += f"   修改为: {issue.get('suggestion','')}"

        # 构造仅供参考的风格建议
        note_list = ""
        if style_note_issues:
            note_list = "\n风格建议（仅参考，改不改由你判断）："
            for issue in style_note_issues:
                note_list += f"\n· {issue.get('description','')}"

        system_prompt = f"""你是一个短视频脚本修改编辑。
修改规则：
1. 只改下面列出的问题，其他地方一个字都不要动
2. 保持原文的语感、节奏、风格
3. 修改后字数与原文基本一致（约{original_duration}字）
4. 只改内容，不改格式标记（如 [快]/[慢]/[重音]）

需要修改的问题：{fix_list}
{note_list}

{format_instructions}"""

        return ChatPromptTemplate.from_messages([
            ("system", system_prompt),
            ("human", "原文脚本：\n\n{content}\n\n请逐条按照上面的修改要求输出新脚本。"),
        ])

    def run(self, state: dict) -> dict:
        """执行修改逻辑。

        Args:
            state: 当前流水线状态

        Returns:
            更新了 script 字段后的 state

        script = state.get("script", {})
        content = script.get("content", "")
        review = state.get("review", {})
        all_issues = review.get("issues", [])

        # 分类问题
        must_fix = [i for i in all_issues if i.get("severity") in ("P0", "P1")]
        style_note = [i for i in all_issues if i.get("severity") == "P2"]

        if not must_fix:
            return state  # 没有要改的

        prompt = self._build_prompt(content, must_fix, style_note)
        chain = prompt | self.llm
        response = chain.invoke({"content": content})

        # 解析新脚本
        try:
            new_script = self.parser.parse(response.content)
        except Exception:
            new_script = Script(
                content=response.content,
                duration_estimate=script.get("duration_estimate", len(content) // 2),
                tone_markers=[],
            )

        state["script"] = new_script.model_dump()
        return state
```

```python
"""
编排器 —— 自定义状态机。
用 while 循环驱动 Agent 顺序执行，管理 step 流转。
Agent 之间通过 state dict 传递数据。
"""

import time
from backend.app.agents.planning import PlanningAgent
from backend.app.agents.writing import WritingAgent
from backend.app.agents.review import ReviewAgent
from backend.app.agents.revision import RevisionAgent

MAX_REVISIONS = 3  # 最大修改轮次


def _build_step_list(state: dict) -> list[dict]:
    """根据当前 state 构建步骤状态列表（供前端展示）。"""
    steps = [
        {"name": "plan", "status": "idle", "label": "策划"},
        {"name": "write", "status": "idle", "label": "写作"},
        {"name": "review", "status": "idle", "label": "审核"},
        {"name": "revise", "status": "idle", "label": "修改"},
    ]
    step_order = ["plan", "write", "review", "revise"]
    current = state.get("step", "plan")

    found_current = False
    for s in steps:
        idx = step_order.index(s["name"])
        if s["name"] == current:
            s["status"] = "active"
            found_current = True
        elif not found_current:
            s["status"] = "done"
    return steps


def run_pipeline(state: dict) -> dict:
    """执行流水线的一步。

    设计要点：
    - while 循环中每次迭代执行一个 step
    - 遇到 wait_confirm 和 done 时 break
    - 这样前端可以多次调用 POST /generate，
      每次只执行一步，跟踪进度

    Args:
        state: 流水线状态字典

    Returns:
        执行一步后的状态字典
    """
    start_time = time.time()

    # 初始化 Agent（首次调用时创建）
    if "_agents" not in state:
        state["_agents"] = {
            "planning": PlanningAgent(),
            "writing": WritingAgent(),
            "review": ReviewAgent(),
            "revision": RevisionAgent(),
        }
        state["revision_count"] = state.get("revision_count", 0)
        if state["step"] not in ("plan", "wait_confirm", "write", "review", "revise", "done"):
            state["step"] = "plan"

    agents = state["_agents"]

    # 状态机主循环 —— 每次只执行一步
    while state["step"] != "done":
        step = state["step"]

        if step == "plan":
            state = agents["planning"].run(state)
            if state.get("auto_mode", True):
                state["step"] = "write"
            else:
                state["step"] = "wait_confirm"
            break  # 一次请求只执行一步

        elif step == "wait_confirm":
            action = state.get("confirm_action", "continue")
            if action == "continue":
                state["step"] = "write"
            elif action == "replan":
                # 重新策划，可选加载用户反馈
                state["outline"] = None
                state["step"] = "plan"
            elif action == "revise_outline":
                # 修改大纲，加载用户反馈
                state["step"] = "plan"
            break  # 暂停，等待下次调用

        elif step == "write":
            state = agents["writing"].run(state)
            state["step"] = "review"
            break

        elif step == "review":
            state = agents["review"].run(state)
            review = state.get("review", {})
            issues = review.get("issues", [])

            # 分离 P0/P1（必改）和 P2（仅记录）
            must_fix = [i for i in issues if i.get("severity") in ("P0", "P1")]

            if not must_fix:
                # 通过 —— 输出终稿
                state["final_script"] = state.get("script")
                state["grade"] = "normal"
                state["step"] = "done"
            elif state.get("revision_count", 0) >= MAX_REVISIONS:
                # 超限 —— 降级输出
                state["final_script"] = state.get("script")
                state["unresolved_issues"] = must_fix
                state["needs_human"] = True
                state["grade"] = "degraded"
                state["step"] = "done"
            else:
                state["step"] = "revise"
            break

        elif step == "revise":
            state = agents["revision"].run(state)
            state["revision_count"] = state.get("revision_count", 0) + 1
            state["step"] = "review"
            break

        else:
            state["step"] = "done"
            break

    # 计算耗时
    elapsed = time.time() - start_time
    state["_elapsed"] = state.get("_elapsed", 0) + elapsed

    # 构建步骤状态列表
    state["_steps"] = _build_step_list(state)

    return state


def init_state(topic: str, style: str = "知识", duration: int = 120,
               auto_mode: bool = True) -> dict:
    """创建初始状态字典。

    Args:
        topic: 选题主题
        style: 账号风格
        duration: 目标时长(秒)
        auto_mode: 是否全自动模式

    Returns:
        初始 state dict
    """
    return {
        "step": "plan",
        "topic": topic,
        "style": style,
        "duration": duration,
        "auto_mode": auto_mode,
        "confirm_action": "",
        "feedback": "",
        "outline": None,
        "script": None,
        "review": None,
        "revision_count": 0,
        "final_script": None,
        "needs_human": False,
        "grade": "normal",
        "unresolved_issues": [],
        "_elapsed": 0,
    }
```

```python
"""
API 路由定义。
POST /generate — 脚本生成（支持全自动 + 人工确认两种模式）
GET  /health   — 健康检查
"""

from fastapi import APIRouter, Depends, HTTPException
from backend.app.schemas.request import GenerateRequest, ConfirmRequest
from backend.app.schemas.response import GenerateResponse, StepInfo
from backend.app.api.deps import get_pipeline
from backend.app.pipeline.orchestator import init_state

router = APIRouter(prefix="")


def _state_to_response(state: dict) -> GenerateResponse:
    """将内部 state dict 转换成 API 响应体。

    Args:
        state: 流水线状态字典

    Returns:
        GenerateResponse 对象
    """
    steps = [
        StepInfo(**s) if isinstance(s, dict) else s
        for s in state.pop("_steps", [])
    ]
    return GenerateResponse(
        step=state.get("step", "plan"),
        steps=steps,
        outline=state.get("outline"),
        script=state.get("script"),
        review=state.get("review"),
        revision_count=state.get("revision_count", 0),
        grade=state.get("grade", "normal"),
        needs_human=state.get("needs_human", False),
        unresolved_issues=state.get("unresolved_issues", []),
        elapsed_time=state.get("_elapsed", 0),
        state=state,  # 完整 state 回传给前端，下次调用时原样传回
    )


@router.post("/generate", response_model=GenerateResponse)
async def generate_script(
    req: GenerateRequest | None = None,
    confirm: ConfirmRequest | None = None,
    pipeline=Depends(get_pipeline),
):
    """生成口播脚本。

    两种调用方式：
    1. 首次调用：传入 GenerateRequest(topic, style, duration, auto_mode)
    2. 确认调用：传入 ConfirmRequest(confirm_action, feedback, state)
       其中 state 是上次调用返回的完整 state

    auto_mode=false 时，策划完成后 paused，返回 wait_confirm 状态。
    前端展示大纲 + 确认按钮，用户确认后再次调用带 confirm 参数。
    """
    # 首次调用
    if req and not confirm:
        if not req.topic.strip():
            raise HTTPException(status_code=422, detail="选题不能为空")
        state = init_state(
            topic=req.topic.strip(),
            style=req.style,
            duration=req.duration,
            auto_mode=req.auto_mode,
        )
    # 确认/继续调用
    elif confirm:
        state = confirm.state
        state["confirm_action"] = confirm.confirm_action
        state["feedback"] = confirm.feedback
        # 清除暂停状态，恢复执行
        if state.get("step") == "wait_confirm":
            pass  # 编排器会读取 confirm_action 决定下一步
        elif state.get("step") == "done":
            return _state_to_response(state)  # 已经完成了，不重复执行
    else:
        raise HTTPException(status_code=422, detail="请传入 GenerateRequest 或 ConfirmRequest")

    # 执行流水线一步
    state = pipeline(state)

    return _state_to_response(state)
```

```python
  })
```
```javascript
  if (!res.ok) {
    const err = await res.json()
    throw new Error(err.detail || '请求失败')
  }
  return res.json()
}
```

```vue
 * 状态流转：
 * idle → running → (confirming) → done
 *
 * 所有数据通过 props 向下传递，子组件通过 emit 向上通知。
 */
import { ref, reactive } from 'vue'
import { generateScript, confirmAction } from './api/index.js'
import InputForm from './components/InputForm.vue'

  script: null,
  review: null,
  revision_count: 0,
  grade: 'normal',
  needs_human: false,
  unresolved_issues: [],
  elapsed_time: 0,
})
// 保存完整 state 用于确认/继续调用
let currentState = null

// 发起生成
async function handleGenerate(params) {
  appState.value = 'running'
  try {
    const data = await generateScript(params)
    updateResponse(data)
  } catch (e) {
    alert('生成失败: ' + e.message)
    appState.value = 'idle'
  }
}

// 确认操作（确认继续/重新策划/修改要求）
async function handleConfirm(action) {
  appState.value = 'running'
  try {
    const data = await confirmAction({
      confirm_action: action.action,
      feedback: action.feedback || '',
      state: currentState,
    })
    updateResponse(data)
  } catch (e) {
    alert('操作失败: ' + e.message)
    appState.value = 'idle'
  }
}

// 重置所有状态
function handleReset() {
  appState.value = 'idle'
  Object.assign(response, {
    step: 'plan', steps: [], outline: null, script: null, review: null,
    revision_count: 0, grade: 'normal', needs_human: false,
    unresolved_issues: [], elapsed_time: 0,
  })
  currentState = null
}

// 更新本地响应数据
function updateResponse(data) {
  Object.assign(response, data)
  currentState = data.state || data

  if (data.step === 'done') {
    appState.value = 'done'
  } else if (data.step === 'wait_confirm') {
    appState.value = 'confirming'
  } else {
    appState.value = 'running'  // 继续下一步
  }
}
</script>
```

```python
"""
评估脚本：单Agent vs 多Agent流水线对比。
对每个选题×风格组合，分别生成单Agent版和多Agent版，
由审核Agent盲评打分，输出对比报告。
"""

import json
import time
from pathlib import Path

# 单Agent：一个 prompt 直接出完整脚本
from langchain_openai import ChatOpenAI
from langchain_core.prompts import ChatPromptTemplate
from backend.app.core.config import get_settings

# 多Agent：走完整流水线
from backend.app.pipeline.orchestator import init_state, run_pipeline


SINGLE_AGENT_PROMPT = """你是一个短视频口播脚本写手。
账号风格：{style}
时长要求：约{duration}秒口语文本
选题：{topic}

请直接输出一个完整口播脚本，包含：
1. 吸引人的开场钩子
2. 核心内容展开
3. 金句收尾

脚本要口语化，能直接读出来。"""


def run_single_agent(topic: str, style: str, duration: int) -> str:
    """单Agent直接写脚本（不经过策划/审核/修改流程）。

    Args:
        topic: 选题
        style: 风格
        duration: 时长

    Returns:
        脚本文本
    """
    settings = get_settings()
    llm = ChatOpenAI(
        model=settings.deepseek_chat_model,
        openai_api_key=settings.deepseek_api_key,
        openai_api_base=settings.deepseek_base_url,
        temperature=0.7,
        max_tokens=4096,
    )
    prompt = ChatPromptTemplate.from_template(SINGLE_AGENT_PROMPT)
    chain = prompt | llm
    response = chain.invoke({
        "topic": topic,
        "style": style,
        "duration": duration,
    })
    return response.content


def run_multi_agent(topic: str, style: str, duration: int) -> dict:
    """多Agent流水线走完整流程。

    Args:
        topic: 选题
        style: 风格
        duration: 时长

    Returns:
        完整 state 字典
    """
    state = init_state(topic=topic, style=style, duration=duration, auto_mode=True)
    state = run_pipeline(state)
    return state


def evaluate_script(content: str, topic: str, style: str) -> dict:
    """用审核Agent单独评估一个脚本。

    Args:
        content: 脚本文本
        topic: 选题
        style: 风格

    Returns:
        评分结果字典
    """
    settings = get_settings()
    llm = ChatOpenAI(
        model=settings.deepseek_reasoner_model,
        openai_api_key=settings.deepseek_api_key,
        openai_api_base=settings.deepseek_base_url,
        temperature=0.3,
        max_tokens=2048,
    )

    eval_prompt = ChatPromptTemplate.from_messages([
        ("system", f"""你是短视频脚本评审专家。给以下脚本打分(0-100)：
- 信息量(25分): 是否有实质性内容
- 口语化(25分): 能否读出嘴
- 合规性(25分): 有无极限词/敏感内容
- 可用率(25分): 能否直接拿去拍

选题: {topic} | 风格: {style}
请只输出一个JSON: {{"score": 85, "dimensions": {{"information": 20, "oral": 20, "compliance": 25, "usability": 20}}}}"""),
        ("human", "{content}"),
    ])

    chain = eval_prompt | llm
    response = chain.invoke({"content": content})

    try:
        return json.loads(response.content.strip().removeprefix("```json").removesuffix("```"))
    except json.JSONDecodeError:
        return {"score": 0, "error": "parse failed"}


def main():
    """主评估流程。"""
    data_path = Path(__file__).parent / "topics.json"
    with open(data_path, "r", encoding="utf-8") as f:
        data = json.load(f)

    print("=" * 60)
    print("评估：单Agent vs 多Agent流水线")
    print(f"数据集：{len(data['topics'])}个选题 × 3种风格 = {len(data['topics']) * 3}个样本")
    print("=" * 60)

    single_scores = []
    multi_scores = []
    results = []

    for item in data["topics"]:
        topic = item["topic"]
        for style in item["styles"]:
            duration = 120
            print(f"\n📝 {topic} | {style} | {duration}s")

            # 单Agent
            t0 = time.time()
            single_script = run_single_agent(topic, style, duration)
            single_elapsed = time.time() - t0
            single_eval = evaluate_script(single_script, topic, style)
            single_score = single_eval.get("score", 0)
            single_scores.append(single_score)
            print(f"  单Agent: {single_score}分 ({single_elapsed:.1f}s)")

            # 多Agent
            t0 = time.time()
            multi_state = run_multi_agent(topic, style, duration)
            multi_elapsed = time.time() - t0
            multi_script = (multi_state.get("final_script") or multi_state.get("script") or {}).get("content", "")
            multi_review = multi_state.get("review", {})
            multi_score = multi_review.get("score", 0)
            multi_scores.append(multi_score)
            revisions = multi_state.get("revision_count", 0)
            print(f"  多Agent: {multi_score}分 ({multi_elapsed:.1f}s, {revisions}轮修改)")

            results.append({
                "topic": topic,
                "style": style,
                "single_score": single_score,
                "multi_score": multi_score,
                "single_elapsed": single_elapsed,
                "multi_elapsed": multi_elapsed,
                "revisions": revisions,
                "grade": multi_state.get("grade", "normal"),
            })

    # 汇总
    print("\n" + "=" * 60)
    print("评估结果汇总")
    print("=" * 60)

    avg_single = sum(single_scores) / len(single_scores) if single_scores else 0
    avg_multi = sum(multi_scores) / len(multi_scores) if multi_scores else 0

    print(f"单Agent平均分: {avg_single:.1f}")
    print(f"多Agent平均分: {avg_multi:.1f}")
    print(f"提升: {avg_multi - avg_single:.1f}分")
    print(f"多Agent通过率: {sum(1 for s in multi_scores if s >= 60) / len(multi_scores) * 100:.0f}%")

    # 按风格分组
    styles = ["知识", "搞笑", "情感"]
    for style in styles:
        style_results = [r for r in results if r["style"] == style]
        if style_results:
            avg_s = sum(r["single_score"] for r in style_results) / len(style_results)
            avg_m = sum(r["multi_score"] for r in style_results) / len(style_results)
            print(f"\n{style}风格 — 单Agent: {avg_s:.1f} | 多Agent: {avg_m:.1f} | 提升: {avg_m - avg_s:.1f}")

    # 保存详细结果
    output_path = Path(__file__).parent / "eval_results.json"
    with open(output_path, "w", encoding="utf-8") as f:
        json.dump({
            "summary": {
                "single_avg": avg_single,
                "multi_avg": avg_multi,
                "improvement": avg_multi - avg_single,
                "multi_pass_rate": sum(1 for s in multi_scores if s >= 60) / len(multi_scores),
            },
            "per_style": {style: ...},
            "details": results,
        }, f, ensure_ascii=False, indent=2)
    print(f"\n详细结果已保存到 {output_path}")


if __name__ == "__main__":
    main()
```

```python
"""
合规检测模块。
提供：极限词检测、医疗敏感词过滤、政治敏感词过滤、Pydantic格式校验。
审核Agent在评分前调用此模块做硬性合规扫描。
"""

import re


# 广告法极限词 —— 正则匹配
FORBIDDEN_WORDS_PATTERN = re.compile(
    r"(最好|第一|国家级|唯一|顶级|极致|绝对|永久|万能|"
    r"首选|冠军|王牌|无敌|史上最强|世界第一|全国第一|"
    r"最便宜|最低价|最先进|最高级|最新科技)",
    re.IGNORECASE
)

# 医疗敏感断言
MEDICAL_CLAIMS_PATTERN = re.compile(
    r"(治愈|根治|神药|特效药|一针见效|药到病除|"
    r"百分百有效|彻底消除|永不复发|祖传秘方|"
    r"抗癌|降三高|排毒养颜)",
    re.IGNORECASE
)

# 政治敏感词（示例库，实际使用需补充）
POLITICAL_SENSITIVE = [
    # 此处为示例，实际使用时填充完整敏感词库
    # 本项目作为演示保留核心入口
]


def detect_forbidden_words(text: str) -> list[dict]:
    """检测广告法极限词。

    Args:
        text: 待检测文本

    Returns:
        检测到的违规词列表，每个违规词包含 {word, position}
    """
    issues = []
    for match in FORBIDDEN_WORDS_PATTERN.finditer(text):
        issues.append({
            "severity": "P0",
            "category": "compliance",
            "word": match.group(),
            "position": match.start(),
            "suggestion": f"删除或替换极限词'{match.group()}'，改用客观描述",
        })
    return issues


def detect_medical_claims(text: str) -> list[dict]:
    """检测医疗健康类敏感断言。

    Args:
        text: 待检测文本

    Returns:
        检测到的违规断言列表
    """
    issues = []
    for match in MEDICAL_CLAIMS_PATTERN.finditer(text):
        issues.append({
            "severity": "P0",
            "category": "compliance",
            "word": match.group(),
            "position": match.start(),
            "suggestion": f"医疗断言'{match.group()}'违规，非专业医疗脚本需删除",
        })
    return issues


def detect_political_sensitive(text: str) -> list[dict]:
    """检测政治敏感内容（基础版）。

    Args:
        text: 待检测文本

    Returns:
        检测到的敏感词列表
    """
    issues = []
    for word in POLITICAL_SENSITIVE:
        pos = text.find(word)
        if pos != -1:
            issues.append({
                "severity": "P0",
                "category": "compliance",
                "word": word,
                "position": pos,
                "suggestion": f"删除敏感词'{word}'",
            })
    return issues


def run_compliance_check(text: str) -> dict:
    """执行完整合规检查。

    Args:
        text: 待检测文本

    Returns:
        {passed: bool, issues: list[dict], summary: str}
    """
    all_issues = []
    all_issues.extend(detect_forbidden_words(text))
    all_issues.extend(detect_medical_claims(text))
    all_issues.extend(detect_political_sensitive(text))

    return {
        "passed": len(all_issues) == 0,
        "issues": all_issues,
        "summary": f"检测到 {len(all_issues)} 个合规问题" if all_issues else "合规检测通过",
    }


def format_check(data: dict, expected_fields: list[str]) -> dict:
    """Pydantic 格式校验。

    校验关键字段是否存在且非空。

    Args:
        data: 待校验的字典
        expected_fields: 必须存在的字段列表

    Returns:
        {passed: bool, missing_fields: list[str]}
    """
    missing = [f for f in expected_fields if not data.get(f)]
    return {
        "passed": len(missing) == 0,
        "missing_fields": missing,
    }
```

```python
"""
LangFuse 可观测追踪。
记录每个Agent的耗时、Token消耗、重试次数。
层1-3使用 print 打日志，层4挂载 LangFuse callback。
"""

import time
import functools
from backend.app.core.config import get_settings


# 模块级开关：有 LangFuse key 才启用
_settings = get_settings()
_LANGFUSE_ENABLED = bool(_settings.langfuse_public_key and _settings.langfuse_secret_key)

if _LANGFUSE_ENABLED:
    try:
        from langfuse import Langfuse
        _langfuse = Langfuse(
            public_key=_settings.langfuse_public_key,
            secret_key=_settings.langfuse_secret_key,
            host=_settings.langfuse_host,
        )
    except Exception:
        _langfuse = None
        _LANGFUSE_ENABLED = False
else:
    _langfuse = None


def trace_agent(agent_name: str):
    """装饰器：追踪 Agent.run() 的执行。

    用法：
        @trace_agent("planning")
        def run(self, state): ...

    Args:
        agent_name: Agent 名称标识

    Returns:
        装饰器函数
    """
    def decorator(func):
        @functools.wraps(func)
        def wrapper(self, state):
            t0 = time.time()

            if _LANGFUSE_ENABLED and _langfuse:
                trace = _langfuse.trace(name=f"agent.{agent_name}")
            else:
                trace = None

            try:
                result = func(self, state)
                elapsed = time.time() - t0

                # 层1-3: print 日志
                print(f"[Agent] {agent_name} | {elapsed:.1f}s | step={state.get('step', '?')}")

                # 层4: LangFuse trace
                if trace:
                    trace.update(
                        metadata={
                            "agent": agent_name,
                            "elapsed": elapsed,
                            "step": state.get("step"),
                            "revision_count": state.get("revision_count", 0),
                        }
                    )

                return result

            except Exception as e:
                elapsed = time.time() - t0
                print(f"[Agent] {agent_name} | {elapsed:.1f}s | ERROR: {e}")
                if trace:
                    trace.update(metadata={"error": str(e), "elapsed": elapsed})
                raise

        return wrapper
    return decorator


def trace_pipeline(step_count: int = 0):
    """追踪整个流水线执行。

    Args:
        step_count: 当前已执行步骤数（用于累积计算）

    Returns:
        trace 对象或 None
    """
    if _LANGFUSE_ENABLED and _langfuse:
        return _langfuse.trace(name="pipeline")
    return None


def flush():
    """刷新 LangFuse 缓冲区（确保数据发送）。"""
    if _LANGFUSE_ENABLED and _langfuse:
        _langfuse.flush()
```
